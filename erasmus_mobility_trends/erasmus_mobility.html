data <- read_rds("erasmus_combined.rds")
data <- data[,-c(1)]

data <- data %>%
  rename(
    sending_city  = `Sending City`,
    receiving_city = `Receiving City`,
    sending_country  = `Sending Country`,
    receiving_country = `Receiving Country`
  )

# Extract ISO-2 codes from labels
data <- data %>%
  mutate(
    send_code = str_extract(sending_country,   "^[A-Z]{2}"),
    recv_code = str_extract(receiving_country, "^[A-Z]{2}")
  )

# Define European analysis universe (programme + key partner countries)
europe_codes <- c(
  "AL","AT","BE","BG","CH","CY","CZ","DE","DK","EE","ES","FI",
  "FR","GR","HR","HU","IE","IS","IT","LI","LT","LU","LV","MT",
  "NL","NO","PL","PT","RO","SE","SI","SK","TR","UK","ME","MK",
  "RS","UA","MD","XK"
)

# Keep only flows where both sending and receiving countries are European
data <- data %>%
  filter(
    send_code %in% europe_codes,
    recv_code %in% europe_codes,
    send_code != recv_code
  )
gdp_data <- read.csv("gdp_data.csv")

# Extract the GDP data for years 2014-2022
gdp_data <- gdp_data[gdp_data$TIME_PERIOD>=2014 & gdp_data$TIME_PERIOD<=2022,]

# Keep only the Euro value
gdp_data <- gdp_data%>%
  filter(
    na_item == 'B1GQ',
    unit == 'CP_MEUR'

gdp_data <- gdp_data[,-c(1, 2, 3, 4, 5,9, 10)]

# Rename columns

# Although the file is saved as tsv the data inside the rows is formatted as CSV meaning that extra transformation is required to extreact the data
pop_data <- read_tsv("pop_count_.tsv")

# Read in the country names
names(pop_data)[1] <- c('country')

# split the file into CSV
pop_data <- pop_data %>%

pop_data <- pop_data[, c(1, 56:64)]

# transform into usable datset
pop_data <- pop_data%>%
  pivot_longer(

    names_to = "year",
    values_to = "population"
  )

pop_data$year <- as.numeric(pop_data$year)
# combine the GDP dataset and the Population dataset by country and year
 add_df <- pop_data%>%
   inner_join(gdp_data, by=c('country', 'year'))

# correct value format

add_df$gdp_value <- as.numeric(add_df$gdp_value)
add_df$population <- as.numeric(add_df$population)
